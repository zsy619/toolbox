package main

import (
    "fmt"
)

func main() {
    start := 1
    end := 10
    
    for i := start; i <= end; i++ {
        fmt.Printf("item_%03d.jpg\n", i)
    }
}
Python 实现 
def generate_sequence(start, end, digits=3, prefix="", suffix=""):
    for i in range(start, end+1):
        formatted_num = str(i).zfill(digits)
        print(f"{prefix}{formatted_num}{suffix}")

if __name__ == "__main__":
    generate_sequence(1, 10, 3, "item_", ".jpg")
